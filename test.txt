.extern my_counter, my_stagod, myNesto
.global handler, globSimbol
.section my_code_handler
.skip 8
handler:
    push %r1
    push %r2
    csrrd %cause, %r1
    ld $2, %r2
    beq %r1, %r2, my_isr_timer
    ld $3, %r2
    beq %r1, %r2, my_isr_terminal
.word nesto
.skip 5
# obrada prekida od tajmera
.section sekcija2
my_isr_timer:
    jmp finish
# obrada prekida od terminala
my_isr_terminal:
    ld my_counter, %r1
    ld $1, %r2
    add %r2, %r1
    st %r1, my_counter
.word 4, 12, my_isr_terminal, 0xF
.skip 5
finish:
    pop %r2
    pop %r1
    iret
.end
.section sekcijaEND
.word 4
