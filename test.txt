.extern my_counter, my_stagod, myNesto
.global handler, globSimbol
.section my_code_handler
.skip 4;
handler:
    push %r1
    push %r2
    csrrd %cause, %r1
    ld $2, %r2
    beq %r1, %r2, my_isr_timer
    ld $3, %r2
    beq %r1, %r2, my_isr_terminal
# obrada prekida od tajmera
.section sekcija2
.skip 16
my_isr_timer:
    jmp finish
# obrada prekida od terminala
my_isr_terminal:
    ld my_counter, %r1
    ld $1, %r2
    add %r2, %r1
    st %r1, my_counter
.word simbol, 4, 3
finish:
    pop %r2
    pop %r1
    iret
.end
